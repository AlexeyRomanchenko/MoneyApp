CREATE TABLE [dbo].[UserRoles] (
    [Role_Id]     INT           IDENTITY (1, 1) NOT NULL,
    [Role]        NVARCHAR (20) NOT NULL,
    [Description] NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Role_Id] ASC)
);
CREATE TABLE [dbo].[Engines] (
    [Engine_Id]     INT           IDENTITY (1, 1) NOT NULL,
    [Engine_type]   NVARCHAR (50) NOT NULL,
    [Turbo]         BIT           NULL,
    [Volume_engine] NVARCHAR (50) NOT NULL,
    [Moment]        NVARCHAR (50) NULL,
    [Capacity]      NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Engine_Id] ASC)
);



CREATE TABLE [dbo].[Cars] (
    [Id]        INT           IDENTITY (1, 1) NOT NULL,
    [Car_name]  NVARCHAR (50) NOT NULL,
    [Engine_Id] INT           NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Cars_Engines] FOREIGN KEY ([Engine_Id]) REFERENCES [dbo].[Engines] ([Engine_Id])
);


CREATE TABLE [dbo].[Users] (
    [Id]       INT           IDENTITY (1, 1) NOT NULL,
    [Username] NVARCHAR (50) NOT NULL,
    [Name]     NVARCHAR (50) NULL,
    [Password] NVARCHAR (50) NOT NULL,
    [Role_Id]  INT           NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Users_UserRoles] FOREIGN KEY ([Role_Id]) REFERENCES [dbo].[UserRoles] ([Role_Id])
);
